36 public object HouseParser : Parser<House?> {
37     @Suppress("MaxLineLength")
38     private val rentedPattern =
39         Regex("""The (?<type>\w+) has been rented by (?<owner>[^.]+)\. (?<pronoun>\w+) has paid the rent until (?<paidUntil>[^.]+)\.""")
40     private val auctionedPattern = Regex("""The (?<type>\w+) is currently being auctioned.""")
41 
42     @Suppress("MaxLineLength")
43     private val bidPattern =
44         Regex("""The auction will end at (?<auctionEnd>[^.]+)\. The highest bid so far is (?<highestBid>\d+) gold and has been submitted by (?<bidder>[^.]+)""")
45     private val moveOutPattern = Regex("""S?[Hh]e will move out on (?<moveDate>[^.]+) \(time of daily""")
46 
47     @Suppress("MaxLineLength")
48     private val transferPattern =
49         Regex("""and (?<action>wants to|will) pass the house to (?<transferee>[\w\s]+) for (?<transferPrice>\d+) gold coin""")

51     override fun fromContent(content: String): House? {
52         val boxContent = boxContent(content)
53         val infoTable = boxContent.selectFirst("table")
54         return house {
55             val (imageRow, descriptionRow) = infoTable.cells()
56             if (imageRow.text().contains("No information about this house found")) return null
57             val imageUrl = imageRow.selectFirst("img")?.attr("src")
58             houseId = imageUrl?.findInteger() ?: throw ParsingException("House image not found")
59             val (title, sizeStr, rentStr, worldStr) = descriptionRow.select("b").map { it.cleanText() }
60             name = title
61             size = sizeStr.remove("square meters").clean().toInt()
62             rent = rentStr.remove("gold").parseThousandSuffix()
63             world = worldStr
64             val descriptionLines = descriptionRow.replaceBrs().wholeText().lines()
65             beds = descriptionLines[1].findInteger()
66             val statusLine = descriptionLines.last().clean()
67             parseStatusDescription(statusLine)
68         }
69     }

71     private fun HouseBuilder.parseStatusDescription(stateLine: String) {
72         rentedPattern.find(stateLine)?.apply {
73             status = HouseStatus.RENTED
74             owner = groups["owner"]!!.value.clean()
75             paidUntil = parseTibiaDateTime(groups["paidUntil"]!!.value)
76             houseType = StringEnum.fromValue("${groups["type"]!!.value}s")
77                 ?: throw ParsingException("Unknown house type found: ${groups["type"]!!.value}")
78         }
79         auctionedPattern.find(stateLine)?.apply {
80             status = HouseStatus.AUCTIONED
81             houseType = StringEnum.fromValue("${groups["type"]!!.value}s")
82                 ?: throw ParsingException("Unknown house type found: ${groups["type"]!!.value}")
83         }
84         bidPattern.find(stateLine)?.apply {
85             status = HouseStatus.AUCTIONED
86             auctionEnd = parseTibiaDateTime(groups["auctionEnd"]!!.value)
87             highestBid = groups["highestBid"]!!.value.parseInteger()
88             highestBidder = groups["bidder"]!!.value.clean()
89         }
90         moveOutPattern.find(stateLine)?.apply {
91             movingDate = parseTibiaDateTime(groups["moveDate"]!!.value)
92         }
93         transferPattern.find(stateLine)?.apply {
94             isTransferAccepted = groups["action"]!!.value == "will"
95             transferRecipient = groups["transferee"]!!.value.clean()
96             transferPrice = groups["transferPrice"]!!.value.toInt()
97         }
98     }
