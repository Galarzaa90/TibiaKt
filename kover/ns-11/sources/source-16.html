40     override fun fromContent(content: String): WorldOverview {
41         val document: Document = Jsoup.parse(content, "", org.jsoup.parser.Parser.xmlParser())
42         val boxContent =
43             document.selectFirst("div.BoxContent") ?: throw ParsingException("BoxContent container not found")
44         val tables = boxContent.select("table.TableContent")
45         val builder = worldOverviewBuilder {
46             recordRegex.find(tables.first()?.text() ?: throw ParsingException("No content tables found."))?.apply {
47                 val (_, recordCount, recordDate) = groupValues
48                 overallMaximumCount = recordCount.parseInteger()
49                 overallMaximumCountDateTime = parseTibiaDateTime(recordDate)
50             }
51             for (i in 1..tables.lastIndex step 2) {
52                 val headerTable = tables[i]
53                 val worldsTable = tables[i + 1]
54                 if (headerTable.text().contains("Regular")) parseWorldsTable(worldsTable)
55             }
56         }
57         return builder.build()
58     }
