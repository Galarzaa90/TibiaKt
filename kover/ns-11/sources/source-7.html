47     private fun EventsScheduleBuilder.parseCalendar(
48         boxContent: Element,
49         yearMonth: YearMonth,
50     ) {
51         var currentMonth = yearMonth
52         val calendarTable = boxContent.selectFirst("#eventscheduletable")
53         var onGoingDay = 1
54         var isFirstDay = true
55         val onGoingEvents: MutableList<EventsScheduleBuilder.EventEntryBuilder> = mutableListOf()
56         for (cell in calendarTable.cells()) {
57             val dayDiv = cell.selectFirst("div") ?: throw ParsingException("could not find day's div")
58             val day = dayDiv.cleanText().toInt()
59             val spans = cell.select("span.HelperDivIndicator")
60             // The calendar might start with a day from the previous month
61             if (onGoingDay < day) currentMonth = currentMonth.minusMonths(1)
62             if (day < onGoingDay) {
63                 currentMonth = currentMonth.plusMonths(1)
64             }
65             onGoingDay = day + 1
66             val todayEvents: MutableList<EventsScheduleBuilder.EventEntryBuilder> = mutableListOf()
67             for (popup in spans) {
68                 val (_, popupContent) = parsePopup(popup.attr("onmouseover"))
69                 val divs = popupContent.select("div")
70                 // Multiple events might be described in the same popup, divs come in pairs, title and content
71                 for (index in 0..divs.lastIndex step 2) {
72                     val event = eventEntryBuilder {
73                         title = divs[index].cleanText().remove(":")
74                         description = divs[index + 1].cleanText().remove("• ")
75                     }
76                     todayEvents.add(event)
77                     // If this is not an event that was already ongoing from a previous day, add to list
78                     if (!onGoingEvents.contains(event)) {
79                         // Only add start date if this is not the first day of the calendar.
80                         // If it's the first day, we have no way to know if the event started that day or before
81                         if (!isFirstDay) {
82                             event.startDate = LocalDate.of(currentMonth.year, currentMonth.month, day)
83                         }
84                         onGoingEvents.add(event)
85                     }
86                 }
87             }
88             // Check which of the ongoing events did not show up today, meaning it has ended now
89             for (pendingEvent in onGoingEvents.toList()) {
90                 if (!todayEvents.contains(pendingEvent)) {
91                     //If it didn't show up today, it means it ended yesterday.
92                     pendingEvent.endDate = LocalDate.of(currentMonth.year, currentMonth.month, day).minusDays(1)
93                     addEntry(pendingEvent.build())
94                     onGoingEvents.remove(pendingEvent)
95                 }
96             }
97             isFirstDay = false
98         }
99         // Add any leftover ongoing events without an end date, as we don't know when they end.
100         onGoingEvents.forEach { addEntry(it.build()) }
101     }
