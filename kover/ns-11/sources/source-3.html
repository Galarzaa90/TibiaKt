38 /** Parser for the CM posts archive. */
39 public object CMPostArchiveParser : Parser<CMPostArchive> {
40     override fun fromContent(content: String): CMPostArchive {
41         val boxContent = HouseParser.boxContent(content)
42         val tables = boxContent.parseTablesMap()
43         return cmPostArchive {
44             tables["CM Post List"]?.let { parsePostList(it) }
45                 ?: throw ParsingException("Could not find CM Post List table.")
46             val form = boxContent.selectFirst("form") ?: throw ParsingException("Could not find search form.")
47 
48             parseSearchTable(form)
49 
50             val paginationData = boxContent.selectFirst("small")?.parsePagination() ?: PaginationData.default()
51             currentPage = paginationData.currentPage
52             totalPages = paginationData.totalPages
53             resultsCount = paginationData.resultsCount
54         }
55     }

57     private fun CMPostArchiveBuilder.parseSearchTable(form: Element) {
58         val formData = form.formData()
59         startDate = LocalDate.of(formData.values["startyear"]?.toInt()
60             ?: throw ParsingException("could not find startyear param"),
61             formData.values["startmonth"]?.toInt() ?: throw ParsingException("could not find startmonth param"),
62             formData.values["startday"]?.toInt() ?: throw ParsingException("could not find startday param"))
63         endDate =
64             LocalDate.of(formData.values["endyear"]?.toInt() ?: throw ParsingException("could not find endyear param"),
65                 formData.values["endmonth"]?.toInt() ?: throw ParsingException("could not find endmonth param"),
66                 formData.values["endday"]?.toInt() ?: throw ParsingException("could not find endday param"))
67     }

69     private fun CMPostArchiveBuilder.parsePostList(table: Element) {
70         for (row in table.rows().offsetStart(1)) {
71             val columns = row.cells()
72             val dateText = columns[0].text()
73             val date = parseTibiaDateTime(dateText)
74             val (forum, thread) = columns[1].replaceBrs().wholeCleanText().split("\n")
75             val postLink =
76                 columns[2]?.selectFirst("a")?.getLinkInformation() ?: throw ParsingException("could not find post link")
77             val postId = postLink.queryParams["postid"]?.get(0)?.toInt()
78                 ?: throw ParsingException("could not find postid in link")
79             addEntry(CMPost(postId, date, forum, thread))
80         }
81     }
