165         }

555     /**
556      * Fetch all the pages, parse and collect the entries.
557      * @param auctionId The id of the auction.
558      * @param type The type of items.
559      * @param paginator The paginator class that holds the collection
560      */
561     private suspend inline fun <reified E, T : AjaxPaginator<E>> fetchAllPages(
562         auctionId: Int,
563         type: AuctionPagesType,
564         paginator: T,
565     ): Pair<Timing, List<E>> {
566         var currentPage = paginator.currentPage + 1
567         val entries: MutableList<E> = paginator.entries.toMutableList()
568         val timing = Timing()
569         while (currentPage <= paginator.totalPages) {
570             val (fetchingTime, ajaxResponse) = fetchAjaxPage(auctionId, type, currentPage)
571             timing.fetching += fetchingTime
572             val parsingTime = measureTimeMillis {
573                 entries.addAll(AuctionParser.parsePageItems(ajaxResponse.ajaxObjects.first().data))
574             } / 1000.0
575             timing.parsing += parsingTime
576             logger.info {
577                 "${
578                     getAuctionAjaxPaginationUrl(auctionId, type, currentPage)
579                 } | PARSE | ${(parsingTime * 1000).toInt()}ms"
580             }
581             currentPage++
582         }
583         return Pair(timing, entries)
584     }
