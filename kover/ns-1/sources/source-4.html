180      * Perform a request using the required headers.
181      * @param method The HTTP method to use for the request.
182      * @param url The URL to request.
183      * @param data The form parameters to add.
184      * @param headers Additional headers to add.
185      */
186     public open suspend fun request(
187         method: HttpMethod,
188         url: String,
189         data: List<Pair<String, Any>> = emptyList(),
190         headers: List<Pair<String, Any>> = emptyList(),
191     ): HttpResponse {
192         val response: HttpResponse = try {
193             when (method) {
194                 HttpMethod.Get -> client.get(url) {
195                     expectSuccess = true
196                     if (headers.isNotEmpty()) {
197                         headers { headers.forEach { header(it.first, it.second) } }
198                     }
199                 }
200 
201                 HttpMethod.Post -> client.submitForm(url, formParameters = Parameters.build {
202                     data.forEach { append(it.first, it.second.toString()) }
203                 }, encodeInQuery = false) {
204 
205                 }
206 
207                 else -> throw IllegalArgumentException("Unsupported method $method")
208             }
209         } catch (re: ResponseException) {
210             if (re.response.status in listOf(
211                     HttpStatusCode.MovedPermanently,
212                     HttpStatusCode.Found,
213                     HttpStatusCode.TemporaryRedirect,
214                     HttpStatusCode.PermanentRedirect,
215                     HttpStatusCode.SeeOther
216                 )
217             ) {
218                 throw SiteMaintenanceException("Tibia.com is under maintenance.", re)
219             }
220             if (re.response.status == HttpStatusCode.Forbidden) {
221                 throw ForbiddenException("403 Forbidden: Might be getting rate-limited", re)
222             }
223             throw NetworkException("${re.response.status.value} ${re.response.status.description}", re)
224         }
225         logger.info {
226             listOf(
227                 url,
228                 method.value.uppercase(),
229                 "${response.status.value} ${response.status.description}",
230                 "${response.fetchingTimeMillis}ms"
231             ).joinToString(" | ")
232         }
233         return response
234     }

691     /**
692      * Fetch all the pages, parse and collect the entries.
693      * @param auctionId The id of the auction.
694      * @param type The type of items.
695      * @param paginator The paginator class that holds the collection
696      */
697     private suspend inline fun <reified E, T : AjaxPaginator<E>> fetchAllPages(
698         auctionId: Int,
699         type: AuctionPagesType,
700         paginator: T,
701     ): Pair<Timing, List<E>> {
702         var currentPage = paginator.currentPage + 1
703         val entries: MutableList<E> = paginator.entries.toMutableList()
704         val timing = Timing()
705         while (currentPage <= paginator.totalPages) {
706             val (fetchingTime, ajaxResponse) = fetchAjaxPage(auctionId, type, currentPage)
707             timing.fetching += fetchingTime
708             val parsingTime = measureTimeMillis {
709                 entries.addAll(AuctionParser.parsePageItems(ajaxResponse.ajaxObjects.first().data))
710             } / 1000.0
711             timing.parsing += parsingTime
712             logger.info {
713                 "${
714                     getAuctionAjaxPaginationUrl(auctionId, type, currentPage)
715                 } | PARSE | ${(parsingTime * 1000).toInt()}ms"
716             }
717             currentPage++
718         }
719         return timing to entries
720     }
