42  * Build a URL to Tibia.com.
43  *
44  * @param section The desired section.
45  * @param params The query arguments to pass.
46  * @param test Whether to get a URL to the testing version of Tibia.com.
47  */
48 public fun buildTibiaUrl(
49     section: String,
50     vararg params: Pair<String, Any?>,
51     test: Boolean = false,
52     anchor: String? = null,
53 ): String {
54     val baseUrl = if (test) "www.test.tibia.com" else "www.tibia.com"
55     return "https://$baseUrl/$section/?${
56         params.filter { it.second != null }.joinToString("&") { (name, value) ->
57             "$name=${URLEncoder.encode(value.toString(), Charsets.ISO_8859_1)}"
58         }
59     }${anchor?.let { "#$it" }.orEmpty()}"
60 }
