50 public fun getLastServerSaveTime(currentTime: ZonedDateTime = ZonedDateTime.now(TIBIA_TIMEZONE)): ZonedDateTime {
51     val serverSaveTime = ZonedDateTime.of(
52         currentTime.toLocalDate(), SERVER_SAVE_TIME, TIBIA_TIMEZONE

77     return try {
