75 internal fun Element.formData(): FormData {
76     if (this.tagName().lowercase() != "form")

82     select("select").forEach {
83         it.select("option").forEach { opt ->
84             val value = opt.attr("value")

86             availableOptions.getOrPut(name) { mutableListOf() }.add(value)
87             if (opt.hasAttr("selected"))
88                 data[name] = value
89         }
90     }
91     select("input[type=checkbox]").forEach {
92         val name = it.attr("name")
93         val value = it.attr("value")
94         availableOptions.getOrPut(name) { mutableListOf() }.add(value)
95         if (it.hasAttr("checked"))
96             dataMultiple.getOrPut(name) { mutableListOf() }.add(value)
97     }
98     select("input[type=radio]").forEach {
99         val name = it.attr("name")
100         val value = it.attr("value")
101         availableOptions.getOrPut(name) { mutableListOf() }.add(value)
102         if (it.hasAttr("checked"))
103             data[name] = value
104     }
105     return FormData(data, dataMultiple, availableOptions, action = attr("action"), method = attr("method"))

129     val (pagesDiv, resultsDiv) = select("small > div")
130     val currentPageLink =
131         pagesDiv.selectFirst(".CurrentPageLink")
132     val pageLinks = pagesDiv.select(".PageLink")
133     val firstOrLastPages = pagesDiv.select(".FirstOrLastElement")
134     val totalPages = if (!firstOrLastPages.isEmpty()) {
135         val lastPageLink = firstOrLastPages.last()?.selectFirst("a")

140         } else {
141             pageLinks[pageLinks.size - 2].text().toInt()
142         }
143     } else {
144         pageLinks.last()?.text()?.toInt() ?: throw ParsingException("could not find last page link")
145     }
146     val page = try {
147         currentPageLink?.text()?.toInt() ?: totalPages
148     } catch (nfe: NumberFormatException) {
149         if (currentPageLink?.text()?.contains("First") == true)

153     }
154     val resultsCount: Int = resultsRegex.find(resultsDiv.text())?.let {
